[데이터 딕셔너리]
 : 사용자와 데이터베이스 자원들을 관리하기 위한 정보들을 저장하는
  시스템 테이블 집합

 - 사용자가 테이블을 생성하거나, 수정하는 모든 정보들이 자동으로 갱신됨
 - 사용자는 데이터 딕셔너리의 정보를 직접 손댈수 없음.
 - 데이터 딕셔너리는 읽기 전용 뷰 형태로 정보를 제공함.

 * 데이터 딕셔너리는 크게 세 가지로 분류됨
 - USER_ : 사용자 계정이 소유한 객체 정보 조회
 - ALL_   : 사용자 계정과 권한을 부여받은 다른 사용자 객체정보 조회
 - DBA_  : 데이터베이스 관리자만 접근 가능한 객체 정보 조회

-- USER_데이터 딕셔너리 **************************************
-- 사용자가 소유한 테이블 정보 조회
SELECT TABLE_NAME FROM USER_TABLES;

-- 사용자가 소유한 제약조건 정보 조회
SELECT * FROM USER_CONSTRAINTS;

-- 사용자가 소유한 시퀀스 정보 조회
SELECT * FROM USER_SEQUENCES;

-- 사용자가 소유한 인덱스 정보 조회
SELECT * FROM USER_INDEXES;

-- 사용자가 소유한 뷰 정보 조회
SELECT * FROM USER_VIEWS;

-- ALL_데이터 딕셔너리 *************************************
-- 사용자가 소유하거나 권한을 부여받은 모든 테이블에 대한 정보 조회
SELECT OWNER, TABLE_NAME FROM ALL_TABLES;

-- DBA_데이터 딕셔너리 ************************************
DBA 권한을 가진 SYSTEM 계정으로 접속해야 함
SELECT OWNER, TABLE_NAME FROM DBA_TABLES;
